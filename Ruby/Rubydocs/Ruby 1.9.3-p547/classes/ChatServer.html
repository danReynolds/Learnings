def add_member(there)

  @mutex.synchronize do
    @members[there] = client
  end
  client

def distribute(there, str)
  name = @members[there].name
  msg = "<#{name}> #{str}"
  msg2 = ">#{name}< #{str}"
  @mutex.synchronize do
    for m in @members.keys
      begin
        if m == there
          @members[m].listen(msg2)
         else
          @members[m].listen(msg)
        end
      rescue
        p $!
        @members.delete(m)
      end
    end
  end
end
