def to_yaml( opts = {} )
    Syck.quick_emit( self, opts ) do |out|
        out.map( "tag:ruby.yaml.org,2002:object:#{ @class }", to_yaml_style ) do |map|
            @ivars.each do |k,v|
                map.add( k, v )
            end
        end
    end
end
