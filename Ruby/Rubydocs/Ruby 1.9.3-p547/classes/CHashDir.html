def add_crl(crl)
  File.open(crl_filename(crl), "w") do |f|
    f << crl.to_pem
  end
  hash_dir(true)

def load_pem_file(filepath)
  str = File.read(filepath)
  begin
    OpenSSL::X509::Certificate.new(str)
  rescue
    begin
      OpenSSL::X509::CRL.new(str)
    rescue
      begin
        OpenSSL::X509::Request.new(str)
      rescue
        nil
      end
    end
  end
end
