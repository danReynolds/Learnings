def initialize(list)
  @list = list
  @check_idx = []
  @list.each_with_index do |x, i|
    @check_idx.push i if x
  end
  @size = @list.size
end

def match(tuple)
  return nil if tuple.size != self.size
  @check_idx.each do |i|
    unless @list[i] === tuple[i]
      return false
    end
  end
  return true
end
