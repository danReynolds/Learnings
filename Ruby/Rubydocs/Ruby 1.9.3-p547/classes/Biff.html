def run
  last = Time.now
  while true
    begin
      sleep(@interval)
      current = File::mtime(@filename)
      if current > last
        changed
        begin
          notify_observers(@filename, current)
        rescue Error
        end
        last = current
      end
    rescue
      next
    end
  end
end
