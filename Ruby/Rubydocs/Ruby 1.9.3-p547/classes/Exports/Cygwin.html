def each_export(objs)
  symprefix = RbConfig::CONFIG["SYMBOL_PREFIX"]
  symprefix.strip! if symprefix
  re = /\s(?:(T)|[[:upper:]])\s#{symprefix}((?!Init_|.*_threadptr_|DllMain\b).*)$/
  objdump(objs) do |l|
    next if /@.*@/ =~ l
    yield $2, !$1 if re =~ l
  end
end
