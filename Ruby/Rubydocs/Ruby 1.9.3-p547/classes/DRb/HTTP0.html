def self.open(uri, config)
  unless /^http:/ =~ uri
    raise(DRbBadScheme, uri) unless uri =~ /^http:/
    raise(DRbBadURI, 'can\t parse uri:' + uri)
  end
  ClientSide.new(uri, config)
end
