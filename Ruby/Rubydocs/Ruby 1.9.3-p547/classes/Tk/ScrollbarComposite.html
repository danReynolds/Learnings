def hscroll(mode, *args)
  st = TkGrid.info(@h_scroll)
  if mode && st.size == 0 then
    @h_scroll.grid('row'=>1, 'column'=>0, 'sticky'=>'ew')
    hook_hscroll_on(*args)
  elsif !mode && st.size != 0 then
    @h_scroll.ungrid
    hook_hscroll_off(*args)
  end
  self
end
