def self.clicks(ms=nil)
  ms = ms.to_s if ms.kind_of?(Symbol)
  case ms
  when nil, ''
    tk_call_without_enc('clock','clicks').to_i
  when /^mic/
    tk_call_without_enc('clock','clicks','-microseconds').to_i
  when /^mil/
    tk_call_without_enc('clock','clicks','-milliseconds').to_i
  else
    tk_call_without_enc('clock','clicks','-milliseconds').to_i
  end
end
