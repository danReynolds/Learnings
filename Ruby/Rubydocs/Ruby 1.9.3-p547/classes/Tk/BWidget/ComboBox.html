def get_listbox(&b)
  win = window(tk_send_without_enc('getlistbox'))
  if b
    if TkCore::WITH_RUBY_VM  ### Ruby 1.9 !!!!
      win.instance_exec(self, &b)
    else
      win.instance_eval(&b)
    end
  end
  win
end
