def self.create_itcl_class(name, body)
  TkCore::INTERP._invoke('::itcl::class', name, body)
  klass = Class.new(Tk::Itcl::ItclObject)
  klass.const_set('ITCL_CLASSNAME', name.dup.freeze)
  klass
end
