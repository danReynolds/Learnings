def get_label(val=nil)
  if val
    tk_call(@path, 'label', val)
  else
    ret = []
    lst = simplelist(tk_call(@path, 'label'))
    while lst.size > 0
      ret << ([num_or_str(lst.shift)] << lst.shift)
    end
  end
end
