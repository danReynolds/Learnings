# File ext/tk/sample/tkcombobox.rb, line 417
def add(ent)
  ent = ent.to_s
  unless @ent_list.index(ent)
    @ent_list << ent
    @lst.insert('end', ent)
  end
  _reset_width
  self
end

def remove(ent)
  ent = ent.to_s
  @ent_list.delete(ent)
  if idx = @lst.get(0, 'end').index(ent)
    @lst.delete(idx)
  end
  _reset_width
  self
end

def values(ary = nil)
  if ary
    @lst.delete(0, 'end')
    @ent_list.clear
    ary.each{|ent| add(ent)}
    _reset_width
    self
  else
    @lst.get(0, 'end')
  end
end
