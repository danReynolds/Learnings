def self.new(chart, keys={})
  obj = nil
  LegenedID_TBL.mutex.synchronize{
    unless (obj = LegenedID_TBL[chart.path])
      (obj = self.allocate).instance_eval{
        @parent = @chart = chart
        @cpath = @chart.path
        @path = @id = 'crosshairs'
        Legend::LegenedID_TBL[@cpath] = self
      }
    end
  }
  chart.legend_configure(keys) if obj && ! keys.empty?
  obj
end
