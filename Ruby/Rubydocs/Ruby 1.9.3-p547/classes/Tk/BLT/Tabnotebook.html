def get_tab(index)
  if (idx = tk_send_without_enc('id', tagindex(index))).empty?
    nil
  else
    Tk::BLT::Tabset::Tab.id2obj(self, idx)
  end
end

def get_tabobj(index)

def insert(pos=nil, keys={})
  if pos.kind_of?(Hash)
    keys = pos
    pos = nil
  end
  pos = 'end' if pos.nil?
  Tk::BLT::Tabnotebook::Tab.new(self, nil,
                                tk_send('insert', tagindex(pos), keys))

end
