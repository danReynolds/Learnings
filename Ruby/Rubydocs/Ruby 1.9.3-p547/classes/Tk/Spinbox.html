def spindown
  begin
    tk_send_without_enc('invoke', 'buttondown')
  rescue RuntimeError => e
    # old version of element?
    begin
      tk_send_without_enc('invoke', 'spinup')
    rescue
      fail e
    end
  end
  self
end
