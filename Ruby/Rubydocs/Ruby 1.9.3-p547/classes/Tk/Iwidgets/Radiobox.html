def add(tag=nil, keys={})
  if tag.kind_of?(Hash)
    keys = tag
    tag = nil
  end
  if tag
    tag = Tk::Itk::Component.new(self, tagid(tag))
  else
    tag = Tk::Itk::Component.new(self)
  end
  tk_call(@path, 'add', tagid(tag), *hash_kv(keys))
  tag
end

def tagid(tagOrId)
  if tagOrId.kind_of?(Tk::Itk::Component)
    tagOrId.name
  else
    #_get_eval_string(tagOrId)
    tagOrId
  end
end
