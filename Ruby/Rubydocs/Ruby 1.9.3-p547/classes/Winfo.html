def TkWinfo.atom(name, win=nil)
  if win
    number(tk_call_without_enc('winfo', 'atom', '-displayof', win,
                               _get_eval_enc_str(name)))
  else
    number(tk_call_without_enc('winfo', 'atom', _get_eval_enc_str(name)))
  end
end
