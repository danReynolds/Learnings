def textbg_info
  info = @text.configinfo(:background)
  if TkComm::GET_CONFIGINFO_AS_ARRAY
    info[0] = 'textbackground'
    info
  else # ! TkComm::GET_CONFIGINFO_AS_ARRAY
    {'textbackground' => info['background']}
  end
end
