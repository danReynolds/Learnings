 assert_dom_equal(

def form_text(action = "/", id = nil, html_class = nil, remote = nil, multipart = nil, method = nil)
  txt =  %Q{<form accept-charset="UTF-8" action="#{action}"}
  txt << %Q{ enctype="multipart/form-data"} if multipart
  txt << %Q{ data-remote="true"} if remote
  txt << %Q{ class="#{html_class}"} if html_class
  txt << %Q{ id="#{id}"} if id
  method = method.to_s == "get" ? "get" : "post"
  txt << %Q{ method="#{method}">}
end

def hidden_fields(method = nil)
  txt =  %Q{<div style="display:none">}
  txt << %Q{<input name="utf8" type="hidden" value="&#x2713;" />}
  if method && !%w(get post).include?(method.to_s)
    txt << %Q{<input name="_method" type="hidden" value="#{method}" />}
  end
  txt << %Q{</div>}
end
