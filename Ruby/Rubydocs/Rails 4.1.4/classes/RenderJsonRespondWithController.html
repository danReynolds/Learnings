def create
  resource = ValidatedCustomer.new(params[:name], 1)
  respond_with(resource) do |format|
    format.json do
      if resource.errors.empty?
        render :json => { :valid => true }
      else
        render :json => { :valid => false }
      end
    end
  end
end
