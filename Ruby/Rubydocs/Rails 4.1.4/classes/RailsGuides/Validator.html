def validate
  validator = MarkupValidator.new
  STDOUT.sync = true
  errors_on_guides = {}

  guides_to_validate.each do |f|
    begin
      results = validator.validate_file(f)
    rescue Exception => e
      puts "\nCould not validate #{f} because of #{e}"
      next
    end

    if results.validity
      print "."
    else
      print "E"
      errors_on_guides[f] = results.errors
    end
  end

  show_results(errors_on_guides)
end
