def test_array
  @request.if_none_match = etag([%w(1 2 3), 'ab', :cde, [:f]])
  get :array
  assert_response :not_modified

  @request.if_none_match = %Q("nomatch")
  get :array
  assert_response :success
end

def test_multiple_etags
