def test_extract_schema_and_table
  {
    %Q(table_name)            => [nil,'table_name'],
    %Q("table.name")          => [nil,'table.name'],
    %Q(schema.table_name)     => %w{schema table_name},
    %Q("schema".table_name)   => %w{schema table_name},
    %Q(schema."table_name")   => %w{schema table_name},
    %Q("schema"."table_name") => %w{schema table_name},
    %Q("even spaces".table)   => ['even spaces','table'],
    %Q(schema."table.name")   => ['schema', 'table.name']
  }.each do |given, expect|
    assert_equal expect, extract_schema_and_table(given)
  end
end
